
package database_console;

import java.sql.*;



public class BoerPietDBConnect {

 
    public static void main(String[] args) {

        try {//tell jvm to load driver
            Class.forName("com.mysql.jdbc.Driver");
            System.out.println("Driver loaded");
            
            //create strings with user info
            String host = "jdbc:mysql://localhost:3306/boerpiet";
            String uName = "root";
            String uPass = "Birdwatching11";
            
            //set up a connection to the database
            Connection con = DriverManager.getConnection(host, uName, uPass);
            System.out.println("Database connected");
            
            //create a statement
            Statement stmt = con.createStatement();
            String SQL = "SELECT voornaam, achternaam, tussenvoegsel FROM klant"
                    + " WHERE tussenvoegsel != 'null'";
            
            //create resultset to contain database info
            ResultSet rs = stmt.executeQuery(SQL);
            
            while(rs.next())
                System.out.println(rs.getString(1) + " " + rs.getString(2)
                + " " + rs.getString(3));
            //close connection
            con.close();
            
        } catch (SQLException | ClassNotFoundException ex) {
            System.out.println(ex.getMessage());
        }
    }
}
